\section{Problem Statement}
We place $N$ nodes randomly in a $d$-dimensional space according to a specified probability distribution. We define a ``collision'' as any instance where the $L_2$ distance between two distinct pairwise difference vectors is less than or equal to a threshold $\epsilon$. 

Let $X_1, X_2, \dots, X_N \in \mathbb{R}^d$ be the positions of the nodes. We want to find:
\begin{enumerate}
    \item The expected probability $P_c$ that at least one collision occurs.
    \item The expected capacity $N$, defined as the number of nodes where $P_c = 0.5$.
\end{enumerate}

\section{General Derivation}
\subsection{Combinatorics and Single Collision Probability}
The total number of difference vectors is $M = \binom{N}{2} \approx \frac{N^2}{2}$. 
To find a collision, we compare all pairs of these difference vectors. The number of such pairs, $K$, is:
\begin{equation}
    K = \binom{M}{2} \approx \frac{M^2}{2} \approx \frac{N^4}{8}
\end{equation}

Let $Z = X_i - X_j - X_k + X_l$. Since $X_i$ are independent and identically distributed, $Z$ represents the sum of four independent draws from the underlying distribution. 
For a small tolerance $\epsilon$, the probability $p$ that a single pair collides is the volume of a $d$-dimensional $\epsilon$-ball multiplied by the probability density of $Z$ at the origin, $f_Z(0)$:
\begin{equation}
    p = P(\|Z\|_2 \le \epsilon) \approx f_Z(0) \cdot V_d(\epsilon) = f_Z(0) \frac{\pi^{d/2}}{\Gamma(d/2 + 1)} \epsilon^d
\end{equation}

\subsection{Expected Number of Collisions and Poisson Approximation}
The expected number of collisions, $\lambda$, is:
\begin{equation}
    \lambda = K \cdot p \approx \frac{N^4}{8} f_Z(0) \frac{\pi^{d/2}}{\Gamma(d/2 + 1)} \epsilon^d
\end{equation}
The probability of experiencing \textit{at least one} collision ($P_c$) follows a Poisson approximation:
\begin{equation}
    P_c \approx 1 - e^{-\lambda}
\end{equation}

\subsection{General Expected Capacity}
We define the expected capacity as the value of $N$ for which $P_c = 0.5$.
\begin{equation}
    1 - e^{-\lambda} = 0.5 \implies e^{-\lambda} = 0.5 \implies \lambda = \ln(2)
\end{equation}
Setting our equation for $\lambda$ equal to $\ln(2)$ and solving for $N$:
\begin{equation}
    \frac{N^4}{8} f_Z(0) \frac{\pi^{d/2}}{\Gamma(d/2 + 1)} \epsilon^d = \ln(2)
\end{equation}
\begin{equation}
    N = \left( \frac{8 \ln(2) \cdot \Gamma(d/2 + 1)}{f_Z(0) \pi^{d/2} \epsilon^d} \right)^{1/4}
\end{equation}

\section{Derivations by Distribution}
By the Central Limit Theorem, the sum of four independent random variables in high dimensions approaches $\mathcal{N}(0, 4\Sigma)$, where $\Sigma$ is the covariance matrix of the spatial distribution.

\subsection{Normally Distributed Nodes}
Let $X \sim \mathcal{N}(0, \sigma^2 I_d)$. Then $Z \sim \mathcal{N}(0, 4\sigma^2 I_d)$.
The density at the origin is:
\begin{equation}
    f_Z(0) = \frac{1}{(8\pi\sigma^2)^{d/2}}
\end{equation}
\textbf{Expected Probability of Collision:}
\begin{equation}
    P_{c, \text{normal}} \approx 1 - \exp\left( - \frac{N^4}{8 \cdot \Gamma(d/2 + 1)} \left( \frac{\epsilon}{\sigma \sqrt{8}} \right)^d \right)
\end{equation}
\textbf{Expected Capacity ($P_c = 0.5$):}
\begin{equation}
    N_{\text{normal}} \approx \left( 8 \ln(2) \cdot \Gamma(d/2 + 1) \right)^{1/4} \left( \frac{\sigma \sqrt{8}}{\epsilon} \right)^{d/4}
\end{equation}

\subsection{Uniformly Distributed Inside a Unit Ball}
For a solid $d$-dimensional ball of radius 1, the variance along any dimension is $\frac{1}{d+2}$. Thus, $Z \approx \mathcal{N}(0, \frac{4}{d+2} I_d)$.
The density at the origin is:
\begin{equation}
    f_Z(0) \approx \left( \frac{d+2}{8\pi} \right)^{d/2}
\end{equation}
\textbf{Expected Probability of Collision:}
\begin{equation}
    P_{c, \text{ball}} \approx 1 - \exp\left( - \frac{N^4}{8 \cdot \Gamma(d/2 + 1)} \left( \frac{\epsilon \sqrt{d+2}}{\sqrt{8}} \right)^d \right)
\end{equation}
\textbf{Expected Capacity ($P_c = 0.5$):}
\begin{equation}
    N_{\text{ball}} \approx \left( 8 \ln(2) \cdot \Gamma(d/2 + 1) \right)^{1/4} \left( \frac{\sqrt{8}}{\epsilon \sqrt{d+2}} \right)^{d/4}
\end{equation}

\subsection{Uniformly Distributed on a Unit Hypersphere}
For the surface of a $d$-dimensional unit sphere ($S^{d-1}$), the variance is $\frac{1}{d}$. By CLT, $Z \approx \mathcal{N}(0, \frac{4}{d} I_d)$.
The density at the origin is:
\begin{equation}
    f_Z(0) \approx \left( \frac{d}{8\pi} \right)^{d/2}
\end{equation}
\textbf{Expected Probability of Collision:}
\begin{equation}
    P_{c, \text{sphere}} \approx 1 - \exp\left( - \frac{N^4}{8 \cdot \Gamma(d/2 + 1)} \left( \frac{\epsilon \sqrt{d}}{\sqrt{8}} \right)^d \right)
\end{equation}
\textbf{Expected Capacity ($P_c = 0.5$):}
\begin{equation}
    N_{\text{sphere}} \approx \left( 8 \ln(2) \cdot \Gamma(d/2 + 1) \right)^{1/4} \left( \frac{\sqrt{8}}{\epsilon \sqrt{d}} \right)^{d/4}
\end{equation}
